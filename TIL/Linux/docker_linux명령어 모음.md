# docker 명령어 모음



```sh
[root@localhost /]# sudo docker ps
[root@localhost /]# sudo docker ps -a
[root@localhost /]# sudo start docker postgres
[root@localhost /]# sudo start docker tomcat
[root@localhost /]# sudo docker ps
[root@localhost /]# netstat -ntlp | grep java
[root@localhost /]# docker exec -it [이미지 이름] /bin/bash
root@localhost:/usr/local/tomcat# cd bin
root@localhost:/usr/local/tomcat/bin# 시작
```
